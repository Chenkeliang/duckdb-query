生成 SQL 预览
模式：透视表
WITH base AS (
  SELECT region, city, year, quarter, order_amount
  FROM sales
  WHERE order_date BETWEEN '2024-01-01' AND '2024-12-31'
)
SELECT *
FROM pivot_table(
  ['base'],
  ['sum(order_amount)', 'avg(order_amount)'],
  ['region', 'city'],
  ['year', 'quarter'],
  [],
  subtotals := 1,
  grand_totals := 1,
  values_axis := 'rows'
);
