结果展示选项

高基数提示：列字段建议设置上限 
WITH base AS (
  SELECT shop_code, status, esr_code
  FROM meepo_logi_10_01
  WHERE status IS NOT NULL
)
SELECT *
FROM pivot_table([
        base,
        [COUNT(esr_code) AS count_esr],
        ["shop_code"],
        ["status"],
        []
    ],
        values_axis:='columns',
        fill_value:='0',
        max_columns:=12
    );
WITH base AS (
  SELECT shop_code, status, esr_code
  FROM meepo_logi_10_01
  WHERE status IS NOT NULL
)
SELECT * FROM base PIVOT(COUNT(esr_code) FOR "status" IN ('待发货','在途','已签收'));

原生透视提示

仅支持 1 个列字段；请填写“列值顺序”或设置“列数量上限（建议 10~12）”。高基数字段做列维度时建议设置上限。
